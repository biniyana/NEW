    async function handleSignup(event) {
      event.preventDefault();
      
      if (!selectedUserType) {
        return;
      }

      const button = event.target.querySelector('button[type="submit"]');
      button.disabled = true;
      button.textContent = 'Creating account...';

      const userData = {
        id: Date.now().toString(),
        user_type: selectedUserType,
        name: document.getElementById('signup-name').value,
        email: document.getElementById('signup-email').value,
        phone: document.getElementById('signup-phone').value,
        address: document.getElementById('signup-address').value,
        created_at: new Date().toISOString()
      };

      const result = await window.dataSdk.create(userData);
      
      if (result.isOk) {
        currentUser = userData;
        currentView = 'dashboard';
        render();
      } else {
        button.disabled = false;
        button.textContent = 'Sign Up';
        
        const errorDiv = document.createElement('div');
        errorDiv.style.backgroundColor = '#fee';
        errorDiv.style.color = '#c00';
        errorDiv.style.padding = '12px';
        errorDiv.style.borderRadius = '8px';
        errorDiv.style.marginTop = '16px';
        errorDiv.className = 'text-sm-custom';
        errorDiv.textContent = 'Failed to create account. Please try again.';
        event.target.appendChild(errorDiv);
        
        setTimeout(() => errorDiv.remove(), 5000);
      }
    }

    function toggleChatbot() {
      isChatbotOpen = !isChatbotOpen;
      const panel = document.getElementById('chatbot-panel');
      const button = document.getElementById('chatbot-button');
      
      if (panel && button) {
        panel.style.display = isChatbotOpen ? 'flex' : 'none';
        button.textContent = isChatbotOpen ? '✕' : '🤖';
      }
    }

    function sendChatMessage(event) {
      event.preventDefault();
      
      const input = document.getElementById('chat-input');
      const message = input.value.trim();
      
      if (!message) return;
      
      chatMessages.push({ sender: 'user', text: message });
      input.value = '';
      
      // Simulate bot response
      setTimeout(() => {
        const botResponses = {
          'how': 'You can navigate through different sections using the sidebar menu. Each section has specific features for managing your recyclables!',
          'rate': 'Current rates vary by material type. Check the Rate List section in the dashboard for detailed pricing.',
          'message': 'You can send messages to junkshops through the Messages tab. Simply select a junkshop and start chatting!',
          'request': 'To create a collection request, go to the Requests section and click "New Request". Fill in your details and submit!',
          'default': 'I can help you with navigation, rates, messaging, and collection requests. What would you like to know more about?'
        };
        
        let response = botResponses.default;
        for (let key in botResponses) {
          if (message.toLowerCase().includes(key)) {
            response = botResponses[key];
            break;
          }
        }
        
        chatMessages.push({ sender: 'bot', text: response });
        render();
      }, 1000);
      
      render();
    }

    async function handleSignup(event) {
      event.preventDefault();
      
      if (!selectedUserType) {
        return;
      }

      const button = event.target.querySelector('button[type="submit"]');
      button.disabled = true;
      button.textContent = 'Creating account...';

      const userData = {
        id: Date.now().toString(),
        user_type: selectedUserType,
        name: document.getElementById('signup-name').value,
        email: document.getElementById('signup-email').value,
        phone: document.getElementById('signup-phone').value,
        address: document.getElementById('signup-address').value,
        created_at: new Date().toISOString()
      };

      const result = await window.dataSdk.create(userData);
      
      if (result.isOk) {
        currentUser = userData;
        currentView = 'dashboard';
        render();
      } else {
        button.disabled = false;
        button.textContent = 'Sign Up';
        
        const errorDiv = document.createElement('div');
        errorDiv.style.backgroundColor = '#fee';
        errorDiv.style.color = '#c00';
        errorDiv.style.padding = '12px';
        errorDiv.style.borderRadius = '8px';
        errorDiv.style.marginTop = '16px';
        errorDiv.className = 'text-sm-custom';
        errorDiv.textContent = 'Failed to create account. Please try again.';
        event.target.appendChild(errorDiv);
        
        setTimeout(() => errorDiv.remove(), 5000);
      }
    }
